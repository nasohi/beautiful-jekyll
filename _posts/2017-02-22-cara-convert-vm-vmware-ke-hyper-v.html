https://www.microsoft.com/en-us/download/details.aspx?id=42497


Run PowerShell as Administrator

PS C:\WINDOWS\system32>  Import-Module "C:\Program Files\Microsoft Virtual Machine Converter\MvmcCmdlet.psd1"

PS C:\WINDOWS\system32> ConvertTo-MvmcVirtualHardDisk -SourceLiteralPath "D:\2016-WINSRV2.vmdk" -VhdType DynamicHardDisk
 -VhdFormat vhdx -DestinationLiteralPath "D:\WINRSRV2-2016"
