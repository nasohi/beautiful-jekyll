--- layout: post title: 'Basic laravel virtualhost nginx configuration ' date: '2018-12-05T15:03:00.002+07:00' author: Nasohi Ciptandani tags: - Laravel modified_time: '2018-12-05T15:03:32.824+07:00' blogger_id: tag:blogger.com,1999:blog-7821506235811301823.post-1559813141720984568 blogger_orig_url: https://nciptandani.blogspot.com/2018/12/basic-laravel-virtualhost-nginx.html --- server {
    listen   80 default_server;

    root /var/www/laravel/public/;
    index index.php index.html index.htm;

    location / {
         try_files $uri $uri/ /index.php$is_args$args;
    }

    # pass the PHP scripts to FastCGI server listening on /var/run/php5-fpm.sock
    location ~ \.php$ {
            try_files $uri /index.php =404;
            fastcgi_pass unix:/var/run/php5-fpm.sock;
            fastcgi_index index.php;
            fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
            include fastcgi_params;
    }
}
