--- layout: post title: Configure Reverse Proxy With Apache2 on Debian 9 date: '2019-07-16T20:38:00.000+07:00' author: Nasohi Ciptandani tags: - Apache2 modified_time: '2019-07-16T20:38:23.530+07:00' blogger_id: tag:blogger.com,1999:blog-7821506235811301823.post-220070856902046294 blogger_orig_url: https://nciptandani.blogspot.com/2019/07/configure-reverse-proxy-with-apache2-on.html --- Load modules
#a2enmod proxy
#a2enmod proxy_http

Virtualhost Configuration

#cd /etc/apache2/sites-available/
#vi myproxyweb

<VirtualHost *:80>
        ServerName mywebsite.id
        ProxyPass       /       http://192.168.1.1:8080/
        ProxyPassReverse   /    http://192.168.1.1:8080/
</VirtualHost>

Disable Default

#a2dissite default

Enable new Virtualhost

#a2ensite myproxyweb

Restart Apache

#systemctl restart apache2

OK!
